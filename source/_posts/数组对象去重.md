#1.需将json转换成js数组对象
```
//var arr = JSON.parse(data)

let arr = [
    { name: "小红", id: 1 },
    { name: "小橙", id: 1 },
    { name: "小黄", id: 4 },
    { name: "小绿", id: 3 },
    { name: "小青", id: 1 },
    { name: "小蓝", id: 4 }
];
```
#2.在方法外定义hash数组，方便调用
```
var hash = [];
function quchong(arr) {
						//  第一步，去重
						// var hash = [];
						for (var i = 0; i < arr.length; i++) {
							for (var j = i + 1; j < arr.length; j++) {
								if (arr[i].id === arr[j].id) {
									++i;
									j = i;
								}
							}
							arr[i].num = 0;
							hash.push(arr[i]);
						}
						// 第二步，统计重复个数
						hash.forEach(item => {
							arr.forEach(dd => {
								if (item.id === dd.id) {
									item.num++
								}
							})
						});
						return hash;
						console.log(hash)
					}

quchong(arr)

console.log(hash)
```
