function getQueryString(name) {
    var reg = new RegExp('(^|&)' + name + '=([^&]*)(&|$)');
    var r = window.location.search.substr(1).match(reg);
    if (r != null) return r[2];

}

var sz = {

    'b': '002',
    'c': '003',
}

由于方法中所得key为局部变量，先在全局进行定义，再在方法中赋值，则可全局使用
var key1;
已知 value为001 获取key
function findKeyByValue(obj, value) {
    for (let key in obj) {
        if (obj[key] === value) {
            key1 = key
        }
    }
    return null
}
